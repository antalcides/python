\section{Python側で定義した関数を\LaTeX 側から使う}
\begin{pycode}
def iseven(n):
    if type(n) is not int:
        return u"{0}は整数ではありません。".format(n)
    elif n % 2 == 0:
        return u"{0}は偶数です。".format(n)
    elif n % 2 == 1:
        return u"{0}は奇数です。".format(n)
\end{pycode}
