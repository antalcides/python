\begin{center}
\textbf{PRINCIPIO DEL MÁXIMO PARA ECUACIONES DIFERENCIALES PARCIALES ELÍ%
PTICAS Y APLICACIONES}\\[0pt]
\end{center}

En este trabajo se estudian principios del máximo para ecuaciones
diferenciales parciales elípticas de segundo orden y sus aplicaciones. Para
operadores diferenciales estrictamente elípticos en forma de divergencia se
estudia el principio del máximo generalizado, y como aplicación se prueba la
existencia y unicidad de la solución del Problema de Dirichlet Generalizado.
Para el principio del máximo clásico se prueba que éste es válido para casos
en donde el coeficiente del término de orden cero de un operador diferencial
elíptico puede tomar valores positivos. Como aplicación de éste resultado se
presentan estimaciones para la solución de la ecuación $\triangle u+\kappa
u=C_{0}$, donde \ $\left| \kappa \right| \lneqq \lambda _{1}$ y $\lambda _{1}
$ es el valor propio principal del operador $-\triangle .$ Por último, se
prueba la simetría esférica y monotonía radial de soluciones positivas, que
tienden a cero en el infinito, de la ecuación de Poisson no-lineal $%
-\triangle u=g(u)$ en $\mathbb{R}^{n}.$\newline

\section{Principio del M\'{a}ximo para Soluciones D\'{e}biles}

En lo siguiente $\Omega $ es un dominio acotado en $\mathbb{R}^{n}$ con $%
\partial \Omega $ de clase $C^{1}$, y se considera el operador diferencial $%
L $ en forma de divergencia 
\begin{equation}
L=\sum_{i,j=1}^{n}D_{i}\left( a^{ij}\left( x\right) D_{j}+b^{i}\left(
x\right) \right) +\sum_{i=1}^{n}c^{i}\left( x\right) D_{i}+d\left( x\right) ,
\tag{1}
\end{equation}
donde

\begin{description}
\item[a)]  $L$ es estrictamente el\'{i}ptico, es decir, existe un n\'{u}mero
positivo $\lambda _{0}$ tal que 
\begin{equation}
\sum_{i,j=1}^{n}a^{ij}\left( x\right) \xi _{i}\xi _{j}\geq \lambda
_{0}\left| \xi \right| ^{2}\text{ para todo }x\in \Omega \text{ y }\xi \in 
\mathbb{R}^{n}.  \tag{2}
\end{equation}

\item[b)]  Los coeficientres $a^{ij},b^{i},c^{i}$ y $d$ $\left(
i,j=1,...,n\right) $ son funciones medibles sobre $\Omega $. Además son 
\textbf{acotadas}, esto es existen constantes positivas $\Lambda $ y $\rho $
tales que para todo $x\in \Omega $ y $\lambda _{0}$ en $\left( 2\right) $ se
cumple 
\begin{equation}
\sum_{i,j=1}^{n}\left| a^{ij}\left( x\right) \right| ^{2}\leq \Lambda
,\lambda _{0}^{-2}\text{, \ \ }\sum_{i=1}^{n}\left( \left| b^{i}\left(
x\right) \right| ^{2}+\left| c^{i}\left( x\right) \right| ^{2}\right)
+\lambda ^{-1}\left| d\left( x\right) \right| \leq \rho ^{2}.  \tag{3}
\end{equation}
\end{description}

\begin{definition}
Si $u\in W^{1,2}(\Omega )$ y $f\in L^{2}\left( \Omega \right) $, $u$ es una
soluci\'{o}n d\'{e}bil (subsoluci\'{o}n d\'{e}bil, supersoluci\'{o}n
d\'{e}bil), de 
\begin{equation*}
Lu=f\text{ \ \ en }\Omega ,
\end{equation*}
si para cada $v\in C_{0}^{1}\left( \Omega \right) $ con $v\geq 0$ se cumple 
\begin{equation}
\mathcal{L}\left( u,v\right) =F\left( v\right) \text{ \ }\left( \leq F\left(
v\right) ,\geq F\left( v\right) \right) ,  \tag{4}
\end{equation}
donde 
\begin{equation}
\mathcal{L}\left( u,v\right) :=\int_{\Omega }\left\{ \sum_{i,j=1}^{n}\left(
a^{ij}D_{j}u+b^{i}u\right) D_{i}v-\left( \sum_{i=1}^{n}c^{i}D_{i}u+du\right)
v\right\} dx.  \tag{5}
\end{equation}
$\mathcal{L}$ se denomina la \textbf{forma bilineal asociada} a $L$, y 
\begin{equation}
F\left( v\right) :=-\int_{\Omega }fvdx\text{ \ }\forall v\in C_{0}^{1}\left(
\Omega \right) .  \tag{6}
\end{equation}
\end{definition}

\begin{definition}
Se dice que $u\in W^{1,2}\left( \Omega \right) $ satisface $u\leq 0$ sobre $%
\partial \Omega $ si su parte positiva $u^{+}:=\max \left\{ u,0\right\} \in
W_{0}^{1,2}\left( \Omega \right) .$ Además $u\geq 0$ sobre $\partial \Omega $
si $-u\leq 0$ sobre $\partial \Omega $ y $u\leq v\in W^{1,2}\left( \Omega
\right) $ sobre $\partial \Omega $ si $u-v\leq 0$ sobre . También para $u\in
W^{1,2}\left( \Omega \right) $ se definen 
\begin{equation*}
\sup_{\partial \Omega }u:=\inf \left\{ k\in \mathbb{R}:u\leq k\text{ sobre }%
\partial \Omega \right\} ,
\end{equation*}
\begin{equation*}
\inf_{\partial \Omega }:=-\sup_{\partial \Omega }\left( -u\right) \text{ y }%
\inf \varnothing =\infty \text{, \ donde }\varnothing \text{ es el conjunto
vacio.}
\end{equation*}
Nótese que $\sum_{i=1}^{n}D_{i}b^{i}+d$ es el coeficiente de $u$ en $\left(
1\right) .$ Para el principio del máximo generalizado supondremos que está
expresión es no positiva en el sentido débil, esto es,
\begin{equation}
\int_{\Omega }\left( dv-\sum_{i=1}^{n}b^{i}D_{i}v\right) dx\leq 0\text{ \ }%
\forall v\geq 0,v\in C_{0}^{1}\left( \Omega \right) .  \tag{7}
\end{equation}
\end{definition}

\begin{theorem}
(Principio del m\'{a}ximo generalizado) Sea $L$ el operador dado en $(1),$
cuyos coeficientes satisfacen $\left( 2\right) ,\left( 3\right) $ y $\left(
7\right) $. Si $u\in W^{1,2}\left( \Omega \right) $ es una soluci\'{o}n
d\'{e}bil de $Lu\geq 0\left( \leq 0\right) $ en $\Omega $, entonces 
\begin{equation*}
\sup_{\Omega }u\leq \sup_{\partial \Omega }u^{+}\text{ \ }\left(
\inf_{\Omega }u\geq \inf_{\partial \Omega }u^{+}\right) .
\end{equation*}
\end{theorem}

\begin{proof}
Sea $Lu\geq 0$. Si $u\in W^{1,2}\left( \Omega \right) $ y $v\in
W_{0}^{1,2}\left( \Omega \right) ,$ según $\left( 7\right) $ podemos
escribir $\mathcal{L}\left( u,v\right) \leq 0$ en la forma 
\begin{eqnarray*}
\int_{\Omega }\left\{
\sum_{i,j=1}^{n}a^{ij}D_{j}uD_{i}v-\sum_{i=1}^{n}\left( b^{i}+c^{i}\right)
vD_{i}u\right\} dx &\leq &\int \left\{ duv-\sum_{i=1}^{n}b^{i}D_{i}\left(
uv\right) \right\} dx \\
&\leq &0,
\end{eqnarray*}
para todo $v\geq 0$ tal que $uv\geq 0$. Luego de $\left( 3\right) $ se sigue
que 
\begin{equation}
\int_{\Omega }\sum_{i,j=1}^{n}a^{ij}D_{j}uD_{i}vdx\leq 2\rho \lambda
_{0}\int_{\Omega }v\left| Du\right| dx,  \tag{8}
\end{equation}
para todo $v\geq 0$ tal que $uv\geq 0$.

Ahora se define $l:=\sup\limits_{\partial \Omega }u^{+}$, y sea $k\in 
\mathbb{R} $ tal que $l\leq k<\sup\limits_{\Omega }u$. Si tal $k$ no existe
entonces el teorema queda demostrado.

Supóngase que tal $k$ existe y definamos 
\begin{equation}
v:=\left( u-k\right) ^{+}.  \tag{9}
\end{equation}
Entonces de $\left( 2\right) ,$ $\left( 8\right) $ y la desigualdad de Hö%
lder resulta que 
\begin{equation}
\left\| \nabla v\right\| _{L^{2}\left( \Omega \right) }\leq 2\rho \left\|
v\right\| _{L^{2}\left( F\right) },\text{ \ donde }F:=\mathbf{supp\ }\nabla
v\subset \mathbf{supp\ }v.  \tag{9}
\end{equation}

Ahora por la desigualdad de Sobolev, existe un $c=c\left( n\right) >0$ tal
que 
\begin{equation}
\left\| v\right\| _{L^{\frac{2n}{n-2}}\left( \Omega \right) }\leq c\left\|
\nabla v\right\| _{L^{2}\left( F\right) }\text{ \ \ para }n>2  \tag{10}
\end{equation}
Por consiguiente $\left( 9\right) $ y $\left( 10\right) $ implican que 
\begin{equation}
m\left( \mathbf{supp\ }\nabla v\right) \geq C^{-n}\text{ \ \ },n>2.  \tag{11}
\end{equation}
Desigualdad similar se obtiene cuando $n=2$ pero $c=c\left( n,\rho ,\left|
\Omega \right| \right) $.

Como $\left( 11\right) $ es independiente del valor de $k$, entonces debe
cumplirse cuando $k\longrightarrow \sup\limits_{\Omega }u$. Pero $v=\left(
u-\sup\limits_{\Omega }u\right) ^{+}=0$ en c.t.p. $x\in \Omega $, lo que
implica $\nabla v=0$ y esto contradice a $\left( 11\right) $. En
consecuencia 
\begin{equation*}
\sup_{\Omega }u\leq l\text{.}
\end{equation*}
El resultado cuando $Lu\leq 0$, se obtiene reemplazando en la conclusi\'{o}n
anterior $u$ por $-u$.
\end{proof}

\begin{corollary}
Si $u\in W_{0}^{1,2}\left( \Omega \right) $ y $Lu=0$ en $\Omega $ en el
sentido d\'{e}bil, entonces $u=0$ en $\Omega $ en el sentido d\'{e}bil.
\end{corollary}

\subsection{Aplicaci\'{o}n: Problema de Dirichlet generalizado}

Con el principio del m\'{a}ximo generalizado (colorario) se sigue que existe
soluci\'{o}n d\'{e}bil \'{u}nica para el problema de Direchlet generalizado.
Adicionalmente en la prueba se utilizan los siguientes teoremas: la
alternativa Fredholm, Lax-Milgram, y Rellich-Kandrackov el cual establece
que 
\begin{equation*}
W_{0}^{1,2}\left( \Omega \right) \subset \subset L^{2}\left( \Omega \right) 
\text{ \ si }n>2,
\end{equation*}
y $\Omega $ es abierto acotado en $\mathbb{R}^{n}$ con $\partial \Omega $ de
clase $C^{1}$.

\begin{definition}
Sean $g$, $f\in L^{2}\left( \Omega \right) $ y $L$ el operador diferencial
definido en $\left( 1\right) $, $u\in W^{1,2}\left( \Omega \right) $ es una 
\textbf{solución débil} de 
\begin{equation*}
\left\{ 
\begin{array}{c}
Lu=f\text{ \ en }\Omega , \\ 
u=g\text{ sobre }\partial \Omega ,
\end{array}
\right. 
\end{equation*}
si $u-g\in W_{0}^{1,2}\left( \Omega \right) $ y además 
\begin{equation}
\mathcal{L}\left( u,v\right) =F\left( v\right) =-\int_{\Omega }fvdx\text{, }%
\forall v\in C_{0}^{1}\left( \Omega \right) ,  \tag{12}
\end{equation}
donde $\mathcal{L}$ viene dado por $\left( 5\right) .$
\end{definition}

\begin{lemma}
Si $I:W_{0}^{1,2}\left( \Omega \right) \longrightarrow \left(
W_{0}^{1,2}\left( \Omega \right) \right) ^{\ast }$, $u\longrightarrow Iu$,
donde 
\begin{equation*}
Iu\left( v\right) :=\int_{\Omega }uvdx\text{, }v\in W_{0}^{1,2}\left( \Omega
\right) .
\end{equation*}
Entonces $I$ es una inmersión compacta.
\end{lemma}

\begin{proof}
Se escribe $I=I_{1}I_{2}$ donde $I_{2}:W_{0}^{1,2}\left( \Omega \right)
\longrightarrow L^{2}\left( \Omega \right) $ es la inyecci\'{o}n compacta
del teorema de Rellich-Kondrackov, y $I_{1}:L^{2}\left( \Omega \right)
\longrightarrow \left( W_{0}^{1,2}\left( \Omega \right) \right) ^{\ast }$, $%
u\longrightarrow I_{1}u$, est\'{a} definido como 
\begin{equation*}
I_{1}u\left( v\right) :=\int_{\Omega }uvdx,\text{ \ \ }v\in
W_{0}^{1,2}\left( \Omega \right) .
\end{equation*}
Claramente se tiene $I_{1}$\ es lineal y continua. Adem\'{a}s $I_{1}$\ es
inyectiva con eso se sigue que $I$\ es una inmersi\'{o}n compacta.
\end{proof}

\begin{theorem}
Sea $L$ el operador diferencial dado en $\left( 1\right) $ con coeficientes
satisfaciendo $\left( 2\right) ,\left( 3\right) $ y $\left( 7\right) $, y $%
g\in W^{1,2}\left( \Omega \right) $. Entonces para cualquier $f\in
L^{2}\left( \Omega \right) $ el \textbf{problema de Dirichlet generalizado} 
\begin{equation*}
\left( P.D.\right) \left\{ 
\begin{array}{c}
Lu=f\text{ \ en }\Omega , \\ 
u=g\text{ sobre }\partial \Omega ,
\end{array}
\right.
\end{equation*}
tiene soluci\'{o}n d\'{e}bil \'{u}nica.
\end{theorem}

\begin{proof}
Si $u$ es una soluci\'{o}n d\'{e}bil de $\left( P.D.\right) $, entonces $%
w:=u-g\in W_{0}^{1,2}\left( \Omega \right) $ y adem\'{a}s para cada $v\in
W_{0}^{1,2}\left( \Omega \right) :$%
\begin{equation*}
\mathcal{L}\left( w,v\right) =\widetilde{F}\left( v\right) ,\text{ \ }%
\widetilde{F}\left( v\right) \in \left( W_{0}^{1,2}\left( \Omega \right)
\right) ^{\ast }.
\end{equation*}
Por tanto se considera en la prueba $g\equiv 0$.

$\mathcal{L}\left( .,.\right) $ es una forma bilineal acotada sobre $%
W_{0}^{1,2}\left( \Omega \right) \times W_{0}^{1,2}\left( \Omega \right) $ y
su operador asociado es $L:W_{0}^{1,2}\left( \Omega \right) \longrightarrow
\left( W_{0}^{1,2}\left( \Omega \right) \right) ^{\ast }$, $Lu:=\mathcal{L}%
\left( u,.\right) $.

Existe un $\sigma _{0}$ tal que
\begin{equation*}
\mathcal{L}_{\sigma _{0}}\left( u,v\right) :=\mathcal{L}\left( u,v\right)
+\sigma _{0}\left( u,v\right) 
\end{equation*}
es coerciva. Luego Lax-Milgram implica que para $F\in \left(
W_{0}^{1,2}\left( \Omega \right) \right) ^{\ast }$ existe un único $u\in
W_{0}^{1,2}\left( \Omega \right) $ tal que 
\begin{equation*}
\mathcal{L}_{\sigma _{0}}\left( u,v\right) =F\left( v\right) \text{ \ para
todo }v\in W_{0}^{1,2}\left( \Omega \right) .
\end{equation*}
La aplicación $L_{\sigma _{0}}:W_{0}^{1,2}\left( \Omega \right)
\longrightarrow \left( W_{0}^{1,2}\left( \Omega \right) \right) ^{\ast }$,
dada por 
\begin{equation*}
(L_{\sigma _{0}}u)\left( v\right) :=\mathcal{L}_{\sigma _{0}}\left(
u,v\right) \text{ \ }\forall v\in W_{0}^{1,2}\left( \Omega \right) 
\end{equation*}
es lineal, acotada uno-uno, sobre y $L_{\sigma _{0}}^{-1}$ también es
acotada.

As\'{i}, para $u\in W_{0}^{1,2}\left( \Omega \right) $ y $F\in \left(
W_{0}^{1,2}\left( \Omega \right) \right) ^{\ast }$ se tienen las ecuaciones
equivalentes: $Lu=F$ y $u-\sigma _{0}L_{\sigma _{0}}^{-1}Iu=L_{\sigma
_{0}}^{-1}F.$

El operador $T:=\sigma _{0}L_{\sigma _{0}}^{-1}I$ es compacto, debido al
lema $7$. Además del colorario $1$ se deduce que $I-T$ es inyectiva. Luego
por la alternativa de Fredholm resulta que 
\begin{equation*}
u-Tu=L_{\sigma _{0}}^{-1}F,
\end{equation*}
tiene única solución $u\in W_{0}^{1,2}\left( \Omega \right) $ para cada $%
F\in \left( W_{0}^{1,2}\left( \Omega \right) \right) ^{\ast },$ y con eso el
teorema queda probado.
\end{proof}

\section{Principio del m\'{a}ximo cl\'{a}sico}

Ahora las funci\'{o}nes $u\in C^{2}\left( \Omega \right) \cap C\left( 
\overline{\Omega }\right) $ y el operador $L$ es 
\begin{equation}
Lu:=\sum_{i,j=1}^{n}a_{ij}\left( x\right)
D_{ij}^{2}u+\sum_{i=1}^{n}b_{i}\left( x\right) D_{i}u+c\left( x\right) u, 
\tag{13}
\end{equation}
donde los coeficientes $a_{ij},b_{i}$ y $c$ est\`{a}n definidos en un
abierto, no vacio, $\Omega $ en $\mathbb{R}^{n}$, $n\geq 2$, y son acotados.
Adem\'{a}s $A=\left[ a_{ij}\left( x\right) \right] $ simetr\'{i}a $\forall
x\in \Omega $ y $L$ es estrictamente el\'{i}ptico, $\left( 2\right) $.

\begin{theorem}
(Principio del m\'{a}ximo d\'{e}bil para $c\leq 0$) si $Lu\geq 0$ $\left(
Lu\leq 0\right) $ en $\Omega $ y $c\leq 0$, entonces 
\begin{equation*}
\max_{\overline{\Omega }}u\leq \max_{\partial \Omega }u^{+}\text{ \ }\left(
\min_{\overline{\Omega }}u\geq \min_{\partial \Omega }u^{-}\right) .
\end{equation*}
\end{theorem}

\begin{lemma}
(de Frontera de Hopf) Supóngase que $\Omega $ satisface la condición de bola
interior en $x_{0}\in \partial \Omega $ y $c\equiv 0$ en $L$ dado en $\left(
13\right) $. Si $u\in C^{2}\left( \Omega \right) \cap C\left( \Omega \cup
\left\{ x_{0}\right\} \right) $ satisface $Lu\geq 0$ en $\Omega $ y $u\left(
x_{0}\right) >u\left( x\right) $ $\forall x\in \Omega $, entonces 
\begin{equation*}
\lim \inf_{t\rightarrow 0^{-}}\frac{u\left( x_{0}+t\nu \right) -u\left(
x_{0}\right) }{t}>0\text{,}
\end{equation*}
donde $\nu $ es la normal exterior a la bola interior en $x_{0}.$

Si en las hipótesis iniciales se cambia $c=0$ en $\Omega $ por $c\leq 0$ en $%
\Omega $ y $u\left( x_{0}\right) \geq 0$, se obtiene la misma conclusión.

Si en las hipótesis iniciales se cambia $c=0$ en $\Omega $ por $u\left(
x_{0}\right) =0$ en $\Omega $, se obtiene la misma conclusión (independiente
del signo de $c$).
\end{lemma}

\begin{theorem}
(Principio del máximo (mínimo) fuerte). $\Omega $ un dominio, no
necesariamente acotado, $u\in C^{2}\left( \Omega \right) \cap C^{2}\left( 
\overline{\Omega }\right) $, no constante, satisface $Lu\geq 0$ $\left(
Lu\leq 0\right) $ en $\Omega $. Si $c=0$, $u$ no alcanza un máximo (mínimo)
en el interior de $\Omega $. Si $c\leq 0$, $u$ no alcanza un máximo no
negativo (mínimo positivo) en el interior de $\Omega $. Independiente del
signo de $c$, $u$ no puede ser cero en un máximo (mínimo) interior.
\end{theorem}

\subsection{Aplicación a la teoría de valores propios}

\begin{theorem}
Dado $\Omega $ abierto acotado, $a_{ij}\in C^{\infty }\left( \overline{%
\Omega }\right) $, $a_{ij}=a_{ji}$, $-L$ estrictamente el\'{i}ptico, donde 
\begin{equation*}
Lu=-\sum_{i,j=1}^{n}\frac{\partial }{\partial x_{i}}\left( a_{ij}\left(
x\right) \frac{\partial }{\partial x_{j}}u\right) ,
\end{equation*}
entonces

\begin{enumerate}
\item[$a)$]  El primer valor propio de $L$, $\lambda _{1}$, es positivo y
además 
\begin{equation}
\lambda _{1}=\left\{ \mathcal{L}\left( u,u\right) :u\in H_{0}^{1}\left(
\Omega \right) ,\text{ }\left\| u\right\| _{L^{2}\left( \Omega \right)
}=1\right\}   \tag{14}
\end{equation}

\item[$b)$]  El mínimo en $\left( 14\right) $ es alcanzado por una función $%
w_{1}$ positiva en $\Omega $, que es solucióin débil de 
\begin{equation}
\left\{ 
\begin{array}{c}
Lw_{1}=\lambda _{1}w_{1}\text{ \ en }\Omega , \\ 
w_{1}=0\text{ \ sobre }\partial \Omega .
\end{array}
\right.   \tag{15}
\end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}
De la teoría de operadores compactos autoadjuntos y la teoría espectral se
sigue que los valores propios de $L$ cumplen 
\begin{equation*}
0<\lambda _{1}\leq \lambda _{2}<\lambda _{3}<...
\end{equation*}
Además existe una base ortonormal de vectores propios de $H_{0}^{1}\left(
\Omega \right) $ y con eso se sigue $a).$

Ahora, si $u\in H_{0}^{1}\left( \Omega \right) $ y $\left\| u\right\|
_{L^{2}\left( \Omega \right) }=1$, $\ u$ es solución débil de $\left(
15\right) $ si y sólo si 
\begin{equation*}
\mathcal{L}\left( u,u\right) =\lambda _{1}.
\end{equation*}
De esta afirmación y la parte $a)$ se deduce que 
\begin{equation*}
\left\{ 
\begin{array}{c}
Lu^{+}=\lambda _{1}u^{+}\text{ \ en }\Omega , \\ 
u^{+}=0\text{ \ sobre }\partial \Omega ,
\end{array}
\right. \text{ \ y \ }\left\{ 
\begin{array}{c}
Lu^{-}=\lambda _{1}u^{-}\text{ \ en }\Omega , \\ 
u^{-}=0\text{ \ sobre }\partial \Omega ,
\end{array}
\right. 
\end{equation*}
en el sentido débil.

De la teoria de la regularidad, el teorema $4$ y dado que $u=u^{+}+u^{-}$ se
concluye $b).$
\end{proof}

\subsection{Un principio del m\'{a}ximo donde $c$ puede der positivo}

Sea $\Omega $ un dominio acotado en $\mathbb{R}^{n}$, $L$ el operador
diferencial definido en $\left( 13\right) $ y $\left( S_{\Omega }\right) $
la condición: 
\begin{equation*}
\left( S_{\Omega }\right) \left\{ 
\begin{array}{c}
\text{Existen constantes }\lambda _{0}\text{ y }\Lambda _{0}\text{ tales que:%
} \\ 
\lambda _{0}\in \xi ^{2}\leq \xi ^{T}a\left( x\right) \xi \leq \Lambda
_{0}\xi ^{2}\text{ para todo }\xi \in \mathbb{R}^{n}, \\ 
\left| b_{i}\left( x\right) \right| \leq \Lambda _{0}\text{ y }-\Lambda
_{0}\leq c\left( x\right) \text{ en }\Omega \text{.}
\end{array}
\right. 
\end{equation*}
Adicionalmente se considera la siguiente definición.

\begin{definition}
Usando la notaci\'{o}n $d\left( x\right) =dist\left( x,\partial \Omega
\right) $, $x\in \Omega $, se dice que el conjunto $\Omega $ satisface la 
\textbf{condici\'{o}n de bola interior uniforme en el sentido fuerte}, si
para alg\'{u}n $r>0$, y todo $x\in \Omega $ con $d\left( x\right) \leq r$ le
corresponde un punto m\'{a}s cercano $y\in \partial \Omega $, $d\left(
x\right) =\left\vert x-y\right\vert $,con la propiedad que $B_{r}\left(
z\right) \subset \Omega $, donde $z=y+\frac{r\left( x-y\right) }{\left\vert
x-y\right\vert }$.
\end{definition}

\begin{theorem}
Sup\'{o}ngase que $\Omega $ satisface la condici\'{o}n de bola interior
uniforme en el sentido fuerte, $L$ dado por $\left( 13\right) $, $\left(
S_{\Omega }\right) $ se cumple, $u\in C^{2}\left( \Omega \right) \cap
C\left( \overline{\Omega }\right) $ satisfaciendo 
\begin{equation*}
Lu\leq 0\text{ \ en }\Omega \text{ \ y }u\geq 0\text{ \ sobre }\partial
\Omega
\end{equation*}
es Lipschitz continua en $\overline{\Omega }$. Si $h\in C^{2}\left( \Omega
\right) \cap C\left( \overline{\Omega }\right) $ satisface 
\begin{equation*}
Lh\leq 0\text{ \ y \ }h>0\text{ en }\Omega \text{,}
\end{equation*}
entonces 
\begin{equation*}
u=\beta h\text{ \ en }\Omega \text{ para un }\beta <0\text{ \ \'{o} \ }%
u\equiv 0\text{ \ en }\Omega \text{ \ \'{o} }u>0\text{ \ en }\Omega \text{.}
\end{equation*}
\end{theorem}

\begin{proof}
Si $u\geq 0$ en $\Omega $, y $L_{0}$ es $L$ con $c\equiv 0$ en $\Omega $, $%
L_{0}u+c^{-}u\leq 0$ y por el principio del mínimo fuerte resulta 
\begin{equation*}
\left( i\right) \text{ }u\equiv 0\text{ \ en }\Omega \text{ \ \ ó \ \ }%
\left( ii\right) \text{ }u>0\text{ \ en }\Omega \text{.}
\end{equation*}
Ahora , sea $\min\limits_{\Omega }u<0$, debido a $\left( S_{\Omega }\right) ,
$ la condicción de bola interior uniforme en el sentido fuerte y las hipó%
tesis sobre $h,$ existe una constante $\delta >0$ tal que 
\begin{equation*}
h\left( x\right) \geq \delta d\left( x\right) \text{ en }\Omega .
\end{equation*}
Luego existe el menor $\theta $ tal que $v:=u+\theta h\geq 0$ en $\Omega .$

Con eso $L_{0}v+c^{-}v\leq 0$ en $\Omega $ y por tanto el principio del
m\'{i}nimo fuerte implica 
\begin{equation*}
\left( iii\right) \text{ }v\equiv 0\text{ \ en }\Omega \text{ \ \ \'{o} \ \ }%
\left( iv\right) \text{ }v>0\text{ \ en }\Omega .
\end{equation*}
Si $\left( iv\right) $ se da, se llega a una contradicci\'{o}n. Luego el
teorema queda probado.
\end{proof}

\subsection{Aplicación: Extensión del principio del máximo}

Se considera el operador el\'{i}ptico sim\'{e}trico 
\begin{equation}
Lu=\sum_{i,j=1}^{n}\frac{\partial }{\partial x_{i}}\left( a_{ij}\left(
x\right) \frac{\partial }{\partial x_{j}}u\right) ,  \tag{16}
\end{equation}
donde $a_{ij}\left( x\right) \in C^{\infty }\left( \overline{\Omega }\right)

\end{equation*}
es verdadera si $c\leq 0$ y falsa si $c\equiv \lambda _{1}.$

\begin{proof}
Es consecuencia del teorema $6$ tomando $h:=w_{1}.$
\end{proof}

\section{Simetr\'{i}a esf\'{e}rica y monoton\'{i}a radial de soluciones
positivas para la ecuacion de Poisson no-lineal en $\mathbb{R}^{n}$}

\begin{lemma}
Sean $R>0,$ $u\in C^{2}\left( \left| x\right| >R\right) \cap C\left( \left|
x\right| \geq R\right) ,$ $u>0$ en $\left| x\right| \geq R$, tendiendo a
cero en el infinito y satisfaciendo 
\begin{equation*}
Lu\equiv \left( \Delta +\sum_{j=1}^{n}b_{j}\left( x\right) \partial
_{j}+c\left( x\right) \right) u\leq 0\text{ \ \ en }\left| x\right| >R,
\end{equation*}
donde $b_{j}=O\left( \left| x\right| ^{1-p}\right) $, $c\left( x\right)
=O\left( \left| x\right| ^{-p}\right) $ en el infinito, $p>2,$ y adem\'{a}s
continuas en $\left| x\right| \geq R$. Entonces, existe un $\mu >0$, tal que 
\begin{equation*}
u\left( x\right) \geq \frac{\mu }{\left| x\right| ^{n-2}}\text{.}
\end{equation*}
\end{lemma}

\begin{proof}
De la función comparación $z=\dfrac{1}{r^{n-2}}+\dfrac{1}{r^{s}},$ donde $%
r=\left| x\right| $ y \linebreak $n-2<s<n-4+p,$ y el principio del máximo dé%
bil se obtiene la conclusión.
\end{proof}

\begin{definition}
Si $x\in \mathbb{R}^{n}$, $x^{\lambda }$ es la \textbf{reflexi\'{o}n de }$x$
con respecto al hiperplano $T_{\lambda }$, definido por 
\begin{equation*}
T_{\lambda }:=\left\{ x\in \mathbb{R}^{n}:x\cdot \gamma =\lambda \right\} 
\text{,}
\end{equation*}
donde $\gamma $ es un vector unitario en $\mathbb{R}^{n}$ y $\lambda >0$. Se
definen tambi\'{e}n 
\begin{equation*}
\Sigma \left( \lambda \right) :=\left\{ x\in \mathbb{R}^{n}:x\cdot \gamma
>\lambda \right\} \text{.}
\end{equation*}
\begin{equation*}
\Sigma ^{\prime }\left( \lambda \right) :=\left\{ x^{\lambda }\in \mathbb{R}%
^{n}:x^{\lambda }=2\left( \lambda -x\cdot \gamma \right) \gamma +x\text{, \
\ }x\in \Sigma \left( \lambda \right) \right\} \text{.}
\end{equation*}
\end{definition}

Sin perdida de generalidad, $\gamma =\mathbf{e}_{1}.$

\begin{lemma}
Sean $u:\mathbb{R}^{n}\rightarrow \mathbb{R}$ es positiva y $C^{2}$ en $%
\mathbb{R}^{n},$ $u=O(\left| x\right| ^{-m}),$ en el infinito, $m>0,$ $b:%
\mathbb{R}^{n}\rightarrow \mathbb{R}$ acotado y $g:\mathbb{R}\rightarrow 
\mathbb{R}$ es tal que en el intervalo $0\leq u\leq u_{0}=\max u$, $%
g=g_{1}+g_{2}$ con $g_{1}\in C^{1}$, $g_{2}$ continua y monótona no
decreciente. $u$ es solución de 
\begin{equation}
\Delta u+b\left( x\right) u_{1}+g\left( u\right) =0\text{ \ \ en\ }\mathbb{R}%
^{n}\text{.}  \tag{17}
\end{equation}
Supóngase que existe un $\lambda \in \left[ 0,\infty \right) $ tal que

\textbf{1}. $b\left( x\right) \geq 0$ para todo $x\in \mathbb{R}^{n}$

\textbf{2}. $u_{1}\left( x\right) \leq 0$ y $u\left( x\right) \leq u\left(
x^{\lambda }\right) $ para todo $x\in \Sigma \left( \lambda \right) $,

\textbf{3}. $u\left( z\right) \neq u\left( z^{\lambda }\right) $ para
alg\'{u}n $z\in \Sigma \left( \lambda \right) $.

Entonces 
\begin{eqnarray*}
u\left( x\right) &<&u\left( x^{\lambda }\right) \text{ \ \ para todo \ }x\in
\Sigma \left( \lambda \right) \text{,} \\
u_{1}\left( x\right) &<&0\text{ \ \ para todo \ }x\in T_{\lambda }\text{.}
\end{eqnarray*}
\end{lemma}

\begin{proof}
Se define $v:\Sigma ^{\prime }\left( \lambda \right) \longrightarrow \mathbb{%
R}$ por $v:=u\left( x^{\lambda }\right) $ y se obtiene que 
\begin{equation}
\Delta v\left( x\right) -b\left( x^{\lambda }\right) v_{1}\left( x\right)
+g\left( v\left( x\right) \right) =0\text{ \ \ }\forall x\in \Sigma ^{\prime
}\left( \lambda \right) \text{.}  \tag{18}
\end{equation}
$w:=v-u$ en $\Sigma ^{\prime }\left( \lambda \right) $ satisface $w\leq 0$ y 
$w\neq 0$. Luego de $\left( 17\right) $, $\left( 18\right) $ y el principio
del m\'{a}ximo fuerte se sigue que 
\begin{equation*}
u\left( x\right) <u\left( x^{\lambda }\right) \text{ \ \ para todo \ }x\in
\Sigma \left( \lambda \right) .
\end{equation*}
Como $w=0$ sobre $T_{\lambda }$, del lema de frontera de Hopf se concluye 
\begin{equation*}
u_{1}\left( x\right) <0\text{ \ \ para todo \ }x\in T_{\lambda }\text{.}
\end{equation*}
\end{proof}

\begin{lemma}
Si $f\in C(\mathbb{R}^{n})$ satisface $f\left( y\right) =O\left( \left|
y\right| ^{-q}\right) $ cerca al infinito, \linebreak $q>n+1$, $u\in C^{2}$
y existe un $C>0$ tal que 
\begin{equation*}
u\left( x\right) =C\int_{\mathbb{R}^{n}}\dfrac{1}{\left| x-y\right| ^{n-2}}%
f\left( y\right) dy\text{, \ \ }x\in \mathbb{R}^{n}\text{,}
\end{equation*}
entonces

\textbf{a)} 
\begin{equation*}
\left| x\right| ^{n-2}u(x)\longrightarrow C\int f\left( y\right) dy\text{, \
\ cuando }\left| x\right| \rightarrow \infty .
\end{equation*}

\textbf{b)} 
\begin{equation*}
\frac{\left| x\right| ^{n}}{x_{1}}u_{1}\left( x\right) \longrightarrow
-\left( n-2\right) C\int f\left( y\right) dy,\text{ \ cuando }\left|
x\right| \rightarrow \infty ,\text{\ con }x_{1}\rightarrow \infty \text{.}
\end{equation*}

\textbf{c) }Si $\left\{ \lambda ^{i}\right\} $ es una sucesi\'{o}n de
n\'{u}meros reales con $\lambda ^{i}\longrightarrow \lambda $ y $\left\{
x^{i}\right\} $ es una sucesi\'{o}n en $\mathbb{R}^{n}$ tendiendo al
infinito, con $x_{1}^{i}<\lambda ^{i}$ $\ $para todo $i\in \mathbb{N}$,
entonces 
\begin{equation*}
\frac{\left| x^{i}\right| ^{n}}{\lambda ^{i}-x_{1}^{i}}\left( u\left(
x^{i}\right) -u\left( x^{i^{\lambda ^{i}}}\right) \right) \longrightarrow
2\left( n-2\right) C\int f\left( y\right) \left( \lambda -y_{1}\right) dy.
\end{equation*}
\end{lemma}

\begin{theorem}
Sea $u$ solución positiva y de clase $C^{2}$ de 
\begin{equation}
-\Delta u=g\left( u\right) \text{ \ \ en\ }\mathbb{R}^{n}\text{, }n\geq 3, 
\tag{19}
\end{equation}
con $u\left( x\right) =O\left( \left| x\right| ^{-m}\right) $ en el
infinito, $m>0$.

Supóngase: \textbf{(i)} En el intervalo $0\leq u\leq u_{0}$ donde $%
u_{0}=\max u$, $g=g_{1}+g_{2}$ con $g_{1}\in C^{1}$, $g_{2}$ continua y monó%
tona no decreciente. \textbf{(ii)} Para algún \linebreak $\alpha >\max
\left\{ \frac{n+1}{m},\frac{2}{m}+1\right\} $, $g\left( u\right) =O\left(
u^{\alpha }\right) $ cerca de $u=0$. Entonces $u\left( x\right) $ es esfé%
ricamente simétrica alrededor de algún punto de $\mathbb{R}^{n}$ y $u_{r}<0$
para $r>0,$ donde $r$ es la coordenada radial alrededor de ese punto. Además 
\begin{equation}
\lim_{\left| x\right| \rightarrow \infty }\left| x\right| ^{n-2}u\left(
x\right) =k>0\text{.}  \tag{20}
\end{equation}
\end{theorem}

\begin{proof}
$\left( 19\right) $ se puede escribir de la forma 
\begin{equation*}
\Delta u+c\left( x\right) u=0,
\end{equation*}
donde $p:=m\left( \alpha -1\right) >2$, y $c\left( x\right) =\frac{g\left(
u\left( x\right) \right) }{u\left( x\right) }=O\left( \left| x\right|
^{-p}\right) .$ Luego exist un $C>0$, tal que 
\begin{equation*}
u=C\int \frac{1}{\left| x-y\right| ^{n-2}}f\left( y\right) dy,
\end{equation*}
donde $f\left( y\right) :=g\left( u\left( y\right) \right) \in C\left( 
\mathbb{R}^{n}\right) $ y $f\left( y\right) =O\left( \left| y\right|
^{-\alpha m}\right) $ en el infinto.

De esto, el lema $6a)$ y el lema $4$ se deduce que 
\begin{equation}
\int f\left( y\right) dy>0.  \tag{21}
\end{equation}
Con eso y el lema $6b)$ resulta: 
\begin{equation}
\text{existe un }\widetilde{\lambda }>0\text{ \ \ tal que \ \ }u_{1}\left(
x\right) <0\text{ para }x_{1}\geq \widetilde{\lambda }\text{.}  \tag{22}
\end{equation}
Ahora, se hace una traslación hasta un punto $\mathbf{q=}\left(
q_{1},...,q_{n}\right) $ en $\mathbb{R}^{n}$ y se fijan las coordenadas
radiales con respecto a éste punto (nuevo origen) de modo que 
\begin{equation}
\int f\left( y\right) y_{j}dy=0,\text{ \ \ }j=1,...,n.  \tag{23}
\end{equation}
Luego $\left( 21\right) ,$ $\left( 22\right) ,$ $\left( 23\right) $ y el
lema $6c)$ se sigue que 
\begin{equation}
\text{Existe un }\lambda _{0}>0\text{ tal que para todo }\lambda \geq
\lambda _{0}:u\left( x\right) >u\left( x^{\lambda }\right) \text{ \ \ si }%
x_{1}<\lambda \text{.}  \tag{24}
\end{equation}
De $\left( 22\right) ,$ $\left( 24\right) $, el lema $5$ y el lema $6c)$
resulta un intervalo máximal $\left( \lambda _{1},\infty \right) ,$ $\lambda
_{1}>0,$ donde $\left( 22\right) $ y $\left( 24\right) $ valen. Luego 
\begin{equation*}
u\left( x\right) \geq u\left( x^{\lambda _{1}}\right) \text{ para\ }%
x_{1}<\lambda _{1}\text{ \ \ y \ \ }u_{1}<0\text{ para }x_{1}>\lambda _{1}%
\text{.}
\end{equation*}
La simetría de \ $u$ con respecto al hiperplano $x_{1}=0$ se sigue el lema $5
$ y con eso el teorema queda probado.
\end{proof}
